

Setup ns:
	ip netns add vtrunk
	ip link add vtrunk-1 type veth peer name vtrunk-1.backend
	ip link set vtrunk-1.backend netns vtrunk


Setup router IP:

	# Interface DHCP server needs static IP
	ip netns exec vtrunk ip addr add 10.0.0.1 dev vtrunk-1.backend
	ip netns exec vtrunk ip link set vtrunk-1.backend up

	# The subnet is 10.0.0.0/24 (range from 10.0.0.1 to 10.0.0.255)
	ip netns exec vtrunk ip route add 10.0.0.0/24 dev vtrunk-1.backend


DHCP server:

	# Start DHCP server (using dnsmasq)
	ip netns exec vtrunk dnsmasq --dhcp-range=10.0.0.2,10.0.0.254,255.255.255.0 --interface=vtrunk-1.backend --no-daemon


DHCP client (new terminal window):

	# Run DHCP client (use a new terminal window)
	dhclient -d vtrunk-1
